Function FindBtm:double;
var
 deltaB, rhs, currBtm: double;
begin
  deltaB:=Bt/10;
  currBtm:=Bt-deltaB;      // Subtract deltaB because deltaB is added first thing in the loop
  repeat
    currBtm:=currBtm+deltaB;
    rhs:=Bt + currBtm*Bamax*VL*(1-fc)*gammaB/(power(power(Bamax,alphaB)+power(gammaB*currBtm,alphaB),1/alphaB));
      if (rhs>currBtm) and (deltaB<0) then
        deltaB:=-abs(currBtm-rhs)
      else
        if (rhs<currBtm) and (deltaB>0) then deltaB:=-abs(currBtm-rhs)
        else deltaB:=abs(currBtm-rhs);
  until (abs(currBtm-rhs)<epsilon);// or (deltaB<epsilon);
  FindBtm:=currBtm;
end;

Procedure Assess_Allometry(VC,vW:double);
begin {allometry}
   {Biomass (DW)}
   Bt:= BC/sigmaC;
   if Bt<1e-6 then Bt:=1e-6;
   {canopy allocation}
   VL:=VC*(1-vW);
   {Root allocation}
   VR:=1-VL;
   {find Ba & Btm = biomass if canopy were full}
   
   Btm:=FindBtm;
   Ba:=Bamax*gammaB*Btm/(power(power(Bamax,alphaB)+power(gammaB*Btm,alphaB),1/alphaB));
   if Ba>Btm then Ba:=Btm;
   {woody biomass}
   BW:=max(0,Btm-Ba);
   {leaf biomass}
   BL:=fc*VL*Ba;
   {leaf area}
   L:=a_sla*BL;
   LW:=a_sdla*WC/sigmaC;
   {full-canopy leaf area}
   L_max:=a_sla*VL*Ba;
   {root biomass}
   BR:=VR*Ba;
   {root length}
   Rl:=BR*asRl;
   {average between-root half distance}
   if Rl>0 then Rd:=sqrt(z/pi/Rl) else Rd:=999;
   if Rd>Rdmax then Rd:=Rdmax;
   {near-root depletion factor}
   betanRd:=sqr(sqr(Rd))-sqr(sqr(Rr))-4*sqr(sqr(Rd))*(ln(Rd)-ln(Rr));
   betanRd:=betanRd/(2*sqr(sqr(Rd)-sqr(Rr)));
   betanRd:=(1+betanRd)/(4*pi); 
end;  {allometry}

Function fCi(P,g,k,Gamma,Ca:double):double;
Var b,c:double;
begin
   if g>0 then
     begin
       b:=(P/g)+k-Ca;
       c:=-(k*Ca+P*Gamma/g);
       fCi:=0.5*(-b+sqrt(sqr(b)-4*c));
     end
   else fCI:=Gamma;
end;

Function photosynthesis(vc,vi,vw:double; var PsC, PsIrr, PsW:double):double;
var dum, fpt, PImax, PCmax, kC, gamma,c_sc0:double;
begin

     {7.775 = mm H2O/m/MPa =rhoa*Cp/(lambda*gamma); c_sc in m/hr}
     {0.000335=1/1.6*(12e-6 gC/umolCO2)*(1000L/m3)*(1/22.4 mol/L)}

   Gamma:=42*exp(9.46*(Ta-25)/(Ta+273.15)); //McMurtrie et al (1992) Eq 18.
   kC:=310*exp(23.956*(Ta-25)/(Ta+273.15)); //McMurtrie et al (1992) Eq 18.
   kC:=kC*(1+200/(155*exp(14.509*(Ta-25)/(Ta+273.15))));//McMurtrie et al (1992) Eq 18. in umol/mol
   
   if (Ta>0) and (Ta<2*Topt) then fpT:= Ta*(2*Topt-Ta)/sqr(Topt)
   else fpT:=0; //Hikosaka et al. 2006 Fig 1 scaled to fpT=1 at Ta = Topt

   UWmax:=gW*fuf*(vW/VR);			      // note vw in call = vW *VC

   UW:= UWmax*(1-exp(-kRl*RL/z))*DL*(Psis-Psi_w)/(-Psi_w); // changed 20 June 2024
   if UW<0 then UW:=0; 			              // note vW in call = vW *VC

   if Dl>0 then c_sc0:=UW/(7.775*Dl*Delta_e) else c_sc0:=0; //c_sc0 units  m/hr
  
   if VL=0 then
     begin
       PImax:=0;
       PCmax:=0;
     end
   else
     begin
        PImax:=gI*(vI/VL)*fpT; 			// note vI in call = vi *VC
        PCmax:=gC*(vC/VL)*fpT;              	// note vC in call = vco2 *VC
     end;

   PsW:=c_sc0*scg*DL*(2*Ca/7)*0.000335;
   PsC:= DL*L*PCmax*((5*Ca/7)-gamma)/(kC+5*Ca/7); 

   PsIrr:= DL*((5*Ca/7)-gamma)/(2*gamma+5*Ca/7);
   PsIrr:= PsIrr*L*PImax/(L+LW)/kI;
   if DL>0 then PsIrr:=PsIrr* ln((HSI+Isw/Dl)/(HSI+Isw*exp(-kI*(L+LW))/Dl))
   else PsIrr:=0;

   dum:=min(PsC,min(PsIrr,PsW));   
                                                 
   if DL>0 then c_sc:=dum/(scg*DL*(2*Ca/7)*0.000335) else c_sc:=0;
                            
   UW:=c_sc*7.775*DL*Delta_e;    
                                                             
   Photosynthesis:=dum;
end;

function uptake(Cbar,vi,km,D,g,Q10,psi:double):double;
var
  V,Croot,Cbar2:double;
begin
   if Rl>0 then
     begin
        Cbar2:=Cbar; if Cbar2<0 then Cbar2:=0;
        V:=g*(vi/VR)*(1-exp(-kRl*RL/z))*power(Q10,Ts/10); //changed 20 June 2024
        Croot:=km-Cbar2-V*betanRd/(D*Rl*psi);
        Croot:=0.5*(-Croot+sqrt(sqr(Croot)+4*km*Cbar2));
        Uptake:=V*Croot/(km+Croot);
     end
   else uptake:=0;
end;{uptake}

function Nfix(vi,g,Q10:double):double;
var dum:double;
begin
  Nfix:=g*(vi/VR)*power(Q10,Ts/10)*(1-exp(-kRl*RL/z))/(1+power(LAInfix/Lcrit, epsilonNfix)); // changed 20 June 2024
end;{Nfix}

function water_tension(W:double):double;
var
   b,psi_i,m,n,theta_i:double;
begin
{Clapp & Hornberger 1978 WRR 14:601 water tension, assume air entry at 0.92 WFPS}
  theta_i:=0.92*phis;
  b:=-ln(psi_f/psi_w)/ln(theta_fp/theta_w);
  psi_i:=psi_f*power(theta_i/theta_fp,-b); 
  m:=(11.5-b)*psi_i/0.0736; 
  n:= 0.84-0.08*b/(11.5-b); 
  if theta<theta_i then water_tension:=psi_f*power(theta/theta_fp,-b)
  else water_tension:= -m*(theta/phis-n)*(theta/phis-1);
end; {water_tension}

chiP:=alphaVP-(alphaVC+alphaVN+alphaVP)*(etaVP*VP)/(etaVC*VC+etaVN*VN*etaVP*VP);
